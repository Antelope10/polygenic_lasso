# Taken from https://github.com/HannahVMeyer/PhenotypeSimulator/blob/master/vignettes/Simulation-and-LinearModel.Rmd


# set directory variables
dir=~/code/polygenic_lasso/data
hapdir=$dir/ALL_1000G_phase1integrated_v3_impute

# hap and legend files should already be loaded

# move to correct directory
cd $hapdir

# generate simulated data in oxford format using resampling
# have to experimentally change 'x'p in dummy var in case of "no suitable disease loci"
# edit -n flag to change num of individuals simulated
list='seq 1 22'
for chr in $list; do
    dummyDL=`sed -n '20000'p $hapdir/ALL_1000G_phase1integrated_v3_chr${chr}_impute.legend | cut -d ' '  -f 2`
    hapgen2 -m $hapdir/genetic_map_chr${chr}_combined_b37.txt -l $hapdir/ALL_1000G_phase1integrated_v3_chr${chr}_impute.legend -h $hapdir/chr${chr}.ceu_subset.hap -o $hapdir/genotypes_chr${chr}_hapgen -n 100 0 -dl $dummyDL 0 0 0 -no_haps_output; done
    




# convert to plink format and prune SNPs 
# check .fam files have correct number of individuals (2kb ~= 100 people for .fam)
> file_list
list='seq 1 22'
for chr in $list; do
	plink --data genotypes_chr${chr}_hapgen.controls --oxford-single-chr $chr --make-bed --out  genotypes_chr${chr}_hapgen.controls
	
	plink --bfile genotypes_chr${chr}_hapgen.controls --indep-pairwise 50kb 5 .99 --out genotypes_chr${chr}_hapgen.controls

    plink --bfile genotypes_chr${chr}_hapgen.controls --extract genotypes_chr${chr}_hapgen.controls.prune.in --make-bed --out genotypes_chr${chr}_hapgen.controls.pruned

	echo -e "genotypes_chr${chr}_hapgen.controls.pruned" >> file_list
done

# Merge chromsome-wide files into a single, genome-wide file
# --allow-no-sex is not in PhenotypeSimulator vignette, but doesn't work without; no missnp generated either
plink --merge-list file_list --make-bed --allow-no-sex --out genotypes_genome_hapgen.controls


# normally has --exclude genotypes_genome_hapgen.controls-merge.missnp, but mine doesn't have it
>file_list_nodup
for chr in `seq 1 22`; do
	plink --bfile genotypes_chr${chr}_hapgen.controls.pruned \
		  --make-bed \
		  --out genotypes_chr${chr}_hapgen.controls.nodup
	echo -e "genotypes_chr${chr}_hapgen.controls.nodup" >> file_list_nodup
done

# final merge (note no duplicate full-genome bim/fam/bed)
plink --merge-list file_list_nodup --allow-no-sex --make-bed --out genotypes_genome_hapgen.controls
